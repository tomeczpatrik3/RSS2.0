-- BUILDING:

	-- GET:
	
		Név:		getAll()
		Url: 		/api/building/
		Feladat:	Visszaadja egy listában az összes adatbázisban található épületet
		
		Név:		getNames()
		Url: 		/api/building/getNames
		Feladat:	Visszaadja egy listában az összes adatbázisban található épület nevét
		
		Név:		findByName()
		Url: 		/api/building/findByName/{name}
		Feladat:	Visszaadja az adott névhez tartozó épületet
		
		Név:		findById()
		Url: 		/api/building/findById/{id}
		Feladat:	Visszaadja az adott idhoz tartozó épületet	
		
	-- POST:

		Név:		createBuilding()
		Url: 		/api/building/createBuilding
		Feladat:	Létrehozza a megfelelő épületet

		Név:		deleteByName()
		Url: 		/api/building/deleteByName/{name}
		Feladat:	Törli az adott névhez tartozó épületet
